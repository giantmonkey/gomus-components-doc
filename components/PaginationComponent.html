Metadata

selector go-pagination
*Specifies a CSS selector that identifies this directive within a template. Supported selectors include `element`, `[attribute]`, `.class`, and `:not()`

Inputs

config

Type: PaginationConfig

Example

<pagination-template #p="paginationApi" [id]="config.id" (pageChange)="config.pageChange.emit($event)" >
  <ng-content select="[go-pagination-back]"></ng-content>
  <ng-container *ngIf="!default">
    <ng-template *ngFor="let item of p.pages;" [ngTemplateOutlet]="template" [ngOutletContext]="{item: item}">
    </ng-template>
  </ng-container>
  <ng-container *ngIf="default">
    <button go-pagination-back> vor </button>
    <button *ngFor="let item of p.pages" [go-pagination-item]="item">
      {{ item.label }}
    </button>
    <button go-pagination-next> später </button>
  </ng-container>
  <ng-content select="[go-pagination-next]"></ng-content>
</pagination-template>
